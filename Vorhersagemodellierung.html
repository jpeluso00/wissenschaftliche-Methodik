Anwendung, Ergebnis und Vorhersage

Wenden Sie hier Ihr Modell an und Interpretieren Sie Ihr Ergebnis. Bei Einzelarbeiten sollte der reine Text (ohne Code, Abbildungen etc.) einen Umfang von ca. 1–2 Seiten haben, bei Gruppenarbeiten einen von ca. 2–4 Seiten.

Modell schätzen:

set.seed(69)

sample <- sample(c(TRUE, FALSE), nrow(encoded_train), replace=TRUE, prob=c(0.8,0.2))

data <- as.matrix(encoded_train[sample, ][, -11])
labels <- as.matrix(encoded_train[sample, ]$vermietungen)

val_data <- as.matrix(encoded_train[!sample, ][, -11])
val_labels <- as.matrix(encoded_train[!sample, ]$vermietungen)
set_random_seed(69, disable_gpu = FALSE)

input <- layer_input(shape = c(10))
output <- input |>
  layer_dropout(rate = 0.1) |>
  layer_dense(units = 32, activation = 'relu') |>
  layer_dense(units = 8, activation = 'relu') |>
  layer_dense(units = 1, activation = 'relu')

model <- keras_model(input, output)

model |> compile(
  optimizer = 'adam',
  loss = 'mse',
  metrics = list('mae')
)

model |> fit(
  data,
  labels,
  epochs = 150,
  batch_size = 32,
  validation_data = list(val_data, val_labels)
)
